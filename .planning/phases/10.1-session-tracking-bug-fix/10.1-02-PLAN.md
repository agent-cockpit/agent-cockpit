<!-- Three-tier lookup pattern from research -->

From 10.1-RESEARCH.md:
```typescript
function getOrCreateSessionId(
  claudeSessionId: string,
  workspace: string,
  db: Database.Database,
  cache: Map<string, string>,
): string {
  // Tier 1: Fast cache lookup
  let uuid = cache.get(claudeSessionId);
  if (uuid) {
    return uuid;
  }

  // Tier 2: Database lookup (recover from restart)
  const row = db.prepare('SELECT session_id FROM claude_sessions WHERE claude_id = ?')
    .get(claudeSessionId) as { session_id: string } | undefined;
  if (row) {
    uuid = row.session_id;
    cache.set(claudeSessionId, uuid);
    return uuid;
  }

  // Tier 3: Create new session
  uuid = randomUUID();
  db.prepare('INSERT INTO claude_sessions (session_id, claude_id, workspace, created_at) VALUES (?, ?, ?, ?)')
    .run(uuid, claudeSessionId, workspace, new Date().toISOString());
  cache.set(claudeSessionId, uuid);
  return uuid;
}
```
