    IMPORTS (use .js extensions — NodeNext resolution):
    ```
    import { useState } from 'react'
    import { useParams } from 'react-router'
    import type { NormalizedEvent } from '@cockpit/shared'
    import { useStore } from '../../store/index.js'
    import { EMPTY_EVENTS } from '../../store/eventsSlice.js'
    ```

    TYPES:
    ```typescript
    interface FileEntry {
      filePath: string
      changeType: 'created' | 'modified' | 'deleted'
      diff?: string
    }
    ```

    HELPERS (pure functions, outside component):

    deriveFileTree(events: NormalizedEvent[]): FileEntry[]
    - Filter events to type === 'file_change'
    - Collapse via Map<string, FileEntry> — last write wins (handles multiple changes same file)
    - Return [...map.values()].sort((a,b) => a.filePath.localeCompare(b.filePath))

    formatElapsed(ms: number): string
    - ms < 60000 → "${Math.floor(ms/1000)}s"
    - else → "${Math.floor(ms/60000)}m ${Math.floor((ms%60000)/1000)}s"

    DiffView({ diff }: { diff: string }):
    - Split on '\n'
    - Map each line to a <div> with data-testid based on first char:
      - line.startsWith('+') && !line.startsWith('+++') → className="text-green-600 bg-green-50" data-testid="diff-line-add"
      - line.startsWith('-') && !line.startsWith('---') → className="text-red-600 bg-red-50" data-testid="diff-line-del"
      - line.startsWith('@') → className="text-blue-500"
      - else → no class
    - Wrap in <pre className="text-xs font-mono overflow-x-auto p-3">

    COMPONENT DiffPanel():
    - const { sessionId } = useParams()
    - const events = useStore((s) => s.events[sessionId!] ?? EMPTY_EVENTS)
    - const session = useStore((s) => sessionId ? s.sessions[sessionId] : undefined)
    - const [selectedFilePath, setSelectedFilePath] = useState<string | null>(null)

    Derived values (computed at render, not stored in Zustand):
    - fileTree = deriveFileTree(events)
    - filesTouched = fileTree.length (collapsed count, not raw event count)
    - finalStatus = session?.status ?? 'unknown'
    - startEvent = events.find(e => e.type === 'session_start')
    - endEvent = [...events].reverse().find(e => e.type === 'session_end')
    - startTime = startEvent ? new Date(startEvent.timestamp).getTime() : null
    - endTime = endEvent ? new Date(endEvent.timestamp).getTime() : session?.lastEventAt ? new Date(session.lastEventAt).getTime() : null
    - elapsedMs = startTime && endTime ? endTime - startTime : null
    - selectedEntry = fileTree.find(f => f.filePath === selectedFilePath) ?? null

    LAYOUT (return JSX):
    ```
    <div className="flex flex-col h-full">
      {/* Summary banner */}
      <div className="flex items-center gap-4 px-4 py-2 border-b border-border text-sm">
        <span>{filesTouched} {filesTouched === 1 ? 'file' : 'files'} changed</span>
        <span className="capitalize">{finalStatus}</span>
        {elapsedMs !== null && <span>{formatElapsed(elapsedMs)}</span>}
      </div>

      {/* Body: file tree + diff view */}
      <div className="flex flex-1 overflow-hidden">
        {/* File tree */}
        <div className="w-64 border-r border-border overflow-y-auto">
          {fileTree.length === 0 ? (
            <div className="p-4 text-sm text-muted-foreground">No files changed</div>
          ) : (
            fileTree.map((entry) => (
              <div
                key={entry.filePath}
                data-testid="file-tree-row"
                onClick={() => setSelectedFilePath(entry.filePath)}
                className={`px-3 py-2 cursor-pointer text-xs font-mono hover:bg-muted/50 ${
                  entry.filePath === selectedFilePath ? 'bg-muted font-semibold' : ''
                }`}
              >
                <div>{entry.filePath.split('/').pop()}</div>
                <div className="text-muted-foreground truncate">{entry.filePath}</div>
                <span className="text-[10px] uppercase">{entry.changeType}</span>
              </div>
            ))
          )}
        </div>

        {/* Diff view */}
        <div className="flex-1 overflow-auto">
          {selectedEntry === null ? (
            <div className="flex items-center justify-center h-full text-sm text-muted-foreground">
              Select a file to view its diff
            </div>
          ) : selectedEntry.diff ? (
            <DiffView diff={selectedEntry.diff} />
          ) : (
            <div className="p-4 text-sm text-muted-foreground">No diff available</div>
          )}
        </div>
      </div>
    </div>
    ```

    - Use .js extensions in all imports (NodeNext resolution)
