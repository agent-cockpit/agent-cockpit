    ```tsx
    import { useState } from 'react'
    import { sendWsMessage } from '../../hooks/useSessionEvents.js'

    function TerminalInputBar({ sessionId }: { sessionId: string }) {
      const wsStatus = useStore((s) => s.wsStatus)
      const [inputText, setInputText] = useState('')
      const isConnected = wsStatus === 'connected'

      function handleSend() {
        if (!inputText.trim() || !isConnected) return
        sendWsMessage({ type: 'terminal_input', sessionId, text: inputText.trim() })
        setInputText('')
      }

      return (
        <div className="shrink-0 border-t border-[var(--color-cockpit-dim)]/40 bg-[var(--color-panel-surface)] flex items-center gap-2 px-3 py-2">
          {/* Connection status dot */}
          <span
            className={`w-1.5 h-1.5 rounded-full shrink-0 ${isConnected ? 'bg-[var(--color-cockpit-green)]' : 'bg-[var(--color-cockpit-red)]'}`}
            aria-hidden
          />
          {/* Prompt glyph */}
          <span className="text-[var(--color-cockpit-cyan)] text-[10px] font-[var(--font-mono-data)] select-none shrink-0">
            &gt;_
          </span>
          {/* Input */}
          <input
            type="text"
            value={inputText}
            onChange={(e) => setInputText(e.target.value)}
            onKeyDown={(e) => { if (e.key === 'Enter') handleSend() }}
            disabled={!isConnected}
            placeholder="enter command..."
            className="flex-1 bg-transparent text-[var(--color-foreground)] text-[11px] font-[var(--font-mono-data)]
                       placeholder:text-[var(--color-cockpit-dim)] outline-none disabled:opacity-40"
          />
          {/* Send button */}
          <button
            onClick={handleSend}
            disabled={!inputText.trim() || !isConnected}
            className="cockpit-btn text-[9px] py-0.5 disabled:opacity-30"
          >
            SEND
          </button>
        </div>
      )
    }
    ```

    In the `InstancePopupHub` component, add `<TerminalInputBar sessionId={selectedSessionId ?? ''} />` as the last element inside `Dialog.Content`, AFTER `</Tabs.Root>` and BEFORE `</Dialog.Content>`.

    The `useState` import already exists — confirm it's imported at the top (InstancePopupHub.tsx does not currently use useState, so add it to the react import).

    `selectedSessionId` is already read from the store — pass it to TerminalInputBar.
